1. Executive Summary

InfraTrack is an enterprise-grade activity and time-tracking portal for the Infrastructure Group, hosted on SharePoint Online (Microsoft 365). It provides a unified platform for six functional teams — Network, Security, Compute, Cloud, Storage, and Database — to log activities, track time, manage team bandwidth, and generate utilization reports integrated with Power BI.

1.1 Business Objectives

  • Centralize infrastructure activity tracking across all functional teams
  • Enable team members to log hours against specific activities with date, type, and category
  • Allow functional managers to define team bandwidth (headcount × hours/month) and factor in vacations
  • Generate actionable utilization and capacity reports for resource planning
  • Provide graphical dashboards for quarterly and monthly planning via Power BI

1.2 Key Capabilities

4.2 SharePoint Service Layer

// src/services/SharePointService.ts
import { sp } from "@pnp/sp/presets/all";
import "@pnp/sp/lists";
import "@pnp/sp/items";

export interface IActivity {
  ID: number;
  Title: string;
  Team: string;
  AssignedTo: { Title: string; EMail: string; };
  ActivityType: string;
  Category: string;
  HoursLogged: number;
  EstimatedHours: number;
  Priority: string;
  Status: string;
  JiraTicket: string;
  StartDate: string;
  DueDate: string;
  Description: string;
}

export class SharePointService {
  // ── Activities ──
  async getActivities(filters?: Partial<IActivity>): Promise<IActivity[]> {
    let query = sp.web.lists
      .getByTitle("InfraActivities")
      .items
      .select("ID,Title,Team,AssignedTo/Title,AssignedTo/EMail,ActivityType,Category,HoursLogged,EstimatedHours,Priority,Status,JiraTicket,StartDate,DueDate")
      .expand("AssignedTo")
      .top(500)
      .orderBy("Created", false);
    if (filters?.team) query = query.filter(`Team eq '${filters.team}'`);
    return await query.get<IActivity[]>();
  }

  async createActivity(data: Partial<IActivity>): Promise<void> {
    await sp.web.lists.getByTitle("InfraActivities").items.add(data);
  }

  async updateActivity(id: number, data: Partial<IActivity>): Promise<void> {
    await sp.web.lists.getByTitle("InfraActivities").items.getById(id).update(data);
  }

  // ── Time Logs ──
  async logTime(entry: ITimeLog): Promise<void> {
    await sp.web.lists.getByTitle("InfraTimeLogs").items.add(entry);
  }

  async getTimeLogs(memberId?: number, month?: string): Promise<ITimeLog[]> {
    let filter = month ? `LogDate ge '${month}-01' and LogDate lt '${month}-32'` : "";
    if (memberId) filter += (filter ? " and " : "") + `MemberId eq ${memberId}`;
    return await sp.web.lists.getByTitle("InfraTimeLogs").items
      .filter(filter).top(5000).get<ITimeLog[]>();
  }

  // ── Bandwidth ──
  async getBandwidth(): Promise<IBandwidth[]> {
    return await sp.web.lists.getByTitle("InfraBandwidth").items.top(100).get();
  }

  // ── Leave ──
  async getLeave(month: string): Promise<ILeave[]> {
    return await sp.web.lists.getByTitle("InfraLeave").items
      .filter(`LeaveFrom le '${month}-31' and LeaveTo ge '${month}-01'`)
      .top(500).get();
  }
}

4.3 Utilization Calculation Service

// src/services/UtilizationService.ts
export interface ITeamUtilization {
  team: string;
  headcount: number;
  totalCapacity: number;    // headcount * bwPerPerson
  vacationHours: number;    // sum of leave hours in period
  netCapacity: number;      // totalCapacity - vacationHours
  loggedHours: number;      // sum of time logs in period
  utilizationPct: number;   // loggedHours / netCapacity * 100
}

export class UtilizationService {
  calculateTeamUtilization(
    team: string,
    bandwidth: IBandwidth,
    timeLogs: ITimeLog[],
    leaves: ILeave[],
    month: string
  ): ITeamUtilization {
    const totalCapacity = bandwidth.Headcount * bandwidth.HoursPerPerson;
    
    // Vacation hours: only count working days (Mon–Fri)
    const vacationHours = leaves
      .filter(l => l.Team === team)
      .reduce((sum, l) => {
        const workDays = this.countWorkingDays(
          new Date(l.LeaveFrom), new Date(l.LeaveTo)
        );
        return sum + (workDays * 8);
      }, 0);
    
    const netCapacity = Math.max(0, totalCapacity - vacationHours);
    
    const loggedHours = timeLogs
      .filter(t => t.Team === team && t.LogDate.startsWith(month))
      .reduce((sum, t) => sum + t.HoursSpent, 0);
    
    const utilizationPct = netCapacity > 0
      ? Math.round((loggedHours / netCapacity) * 100)
      : 0;
    
    return { team, headcount: bandwidth.Headcount, totalCapacity, vacationHours, netCapacity, loggedHours, utilizationPct };
  }

  private countWorkingDays(start: Date, end: Date): number {
    let count = 0;
    const cur = new Date(start);
    while (cur <= end) {
      const day = cur.getDay();
      if (day !== 0 && day !== 6) count++; // Skip Sat/Sun
      cur.setDate(cur.getDate() + 1);
    }
    return count;
  }
}

8.2 Item-Level Permissions

  • InfraTimeLogs: Members can only edit their own time entries (use item-level security: "Users can only read and edit their own items")
  • InfraActivities: All team members can create; only AssignedTo or their Manager can edit/delete
  • InfraBandwidth: Manager-only write; all can read

9.2 Key DAX Measures

-- Utilization %
Team Utilization % =
DIVIDE(
    [Total Logged Hours],
    [Net Capacity Hours],
    0
) * 100

-- Net Capacity (deducting vacation)
Net Capacity Hours =
SUMX(
    InfraBandwidth,
    InfraBandwidth[TotalCapacity] - 
    CALCULATE(
        SUM(InfraLeave[HoursImpact]),
        FILTER(InfraLeave, InfraLeave[Team] = InfraBandwidth[Title])
    )
)

-- Monthly Hours per Member
Member Hours MTD =
CALCULATE(
    SUM(InfraTimeLogs[HoursSpent]),
    DATESMTD(InfraTimeLogs[LogDate])
)

-- Activity Completion %
Activity Completion % =
DIVIDE(
    SUM(InfraActivities[HoursLogged]),
    SUM(InfraActivities[EstimatedHours]),
    0
) * 100

10. Utilization Calculation Logic

10.1 Formula Reference

MetricFormulaExample
Total CapacityHeadcount × HoursPerPerson4 × 160 = 640h
Vacation HoursΣ (WorkingDays × 8) for all leaves in month5 days × 8 = 40h
Net CapacityTotal Capacity − Vacation Hours640 − 40 = 600h
Utilization %(Logged Hours ÷ Net Capacity) × 100(528 ÷ 600) × 100 = 88%
Available HoursNet Capacity − Logged Hours600 − 528 = 72h
Per-Member Util.(Member Logged Hours ÷ Member BW) × 100(130 ÷ 160) × 100 = 81%

12. Maintenance & Operations

12.1 Monthly Operations Checklist

  • □ Verify all team bandwidth configs are updated for the new month
  • □ Confirm vacation/leave entries are current before utilization reporting
  • □ Power BI dataset refresh is running on schedule
  • □ Review over-utilized teams and plan redistribution
  • □ Archive completed activities older than 6 months to a separate archive list
  • □ Review and prune custom field definitions that are no longer in use

12.2 Backup & Recovery

  • SharePoint Online lists are backed up by Microsoft (M365 retention policies)
  • Enable SharePoint Versioning on all lists (keep last 50 versions)
  • Enable Recycle Bin retention for 90 days
  • Monthly export to Excel/CSV stored in SharePoint Document Library for long-term audit trail

12.3 Performance Guidance
